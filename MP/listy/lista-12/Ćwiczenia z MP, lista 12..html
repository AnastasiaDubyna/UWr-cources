(define (square x)
	(* x x))
	
(define (abs x)
  (if (< x 0)
      (- 0 x)
      x))

define (average a b)
  (/ (+ a b) 2))
  
(define (improve guess x)
  (average guess (/ x guess)))

(define (sqrt-iter guess x)
  (if (good-enough? guess x)
      guess
      (sqrt-iter (improve guess x) x)))

(define (good-enough? guess x)
  (< (abs (- (square guess)
             x))
     0.001))
	 
(define (sqrt x)
  (sqrt-iter 1.0 x))
  
(define epsilon 0.0001)
(define (is-number-near tested-number goal-number)
  (if (<= (abs (- tested-number goal-number)) epsilon)
      #t
      #f))

  (vector3 [x : Number] [y : Number] [z : Number]))

	(cond
		[(vector2? vect) (sqrt (+ (square (vector2-x vect)
		(square (vector2-y vect)))))]
		[(vector3? vect) (sqrt (+ (square (vector3-x vect)
		(+ (square (vector3-y vect))
			(square (vector3-z vect))))))]))

	(type-case Vector vect
	[(vector2 x y)
	(sqrt (+ (square x) (square y)))]
	[(vector3 x y z)

		(+ (square y)
		(square z))))]))

(define (sqrt x)
  
  (local [(define (improve guess)
    (/ (+ guess (/ x guess)) 2))]

  (local [(define (good-enough? guess)
    (< (abs (- (* guess guess)
               x))
       0.001))]

  (local [(define (sqrt-iter guess)
    (if (good-enough? guess)
        guess
        (sqrt-iter (improve guess))))]
  
  (sqrt-iter 1.0)))))

(define-type ArithExpr
  (const [val : Number])
  (binop [op : Symbol] [l : ArithExpr] [r : ArithExpr])
  (var-expr [id : Symbol])
  (let-expr [id : Symbol] [e1 : ArithExpr] [e2 : ArithExpr]))

(define (binary-operator? op)
  (member op '(+ - * /)))

(define (parse q)
  (cond
    [(s-exp-number? q) (const (s-exp->number q))]
    [(s-exp-symbol? q) (var-expr (s-exp->symbol q))]
    [(s-exp-list? q)
     (let ([ql (s-exp->list q)])
       (cond
         [(and (= (length ql) 3)
               (equal? (first ql) `let))
          (let ([ll (s-exp->list (second ql))])
            (let-expr (s-exp->symbol (first ll))
                      (parse (second ll))
                      (parse (third ql))))]
         [(and (= (length ql) 3)
               (binary-operator? (s-exp->symbol (first ql))))
          (binop (s-exp->symbol (first ql))
                 (parse (second ql))
                 (parse (third ql)))]))]))

(define (op->proc op)
  (case op ['+ +] ['- -] ['* *] ['/ /]))
  
(define (eval e)
    (type-case ArithExpr e
        [(const n) n]
        [(binop op l r) ((op->proc op) (eval l) (eval r))]
        [(var-expr id) (error 'eval "Unbound variable")]
        [(let-expr x e1 e2)
        (eval (subst (const (eval e1)) x e2))]))
        
(define (subst e1 x e2)
    (type-case ArithExpr e2
        [(var-expr y) (if (eq? x y) e1 (var-expr y))]
        [(const n) (const n)]
        [(binop op l r)
        (binop op (subst e1 x l) (subst e1 x r))]
        [(let-expr y e3 e4)
        (let-expr y (subst e1 x e3)
                    (if (eq? x y)
                        e4
                        (subst e1 x e4)))]))
                        
(test (eval (parse `(let (x 3) (+ x x)))) 6)
(test (eval (parse `(let (x 3) (let (x 2) (+ x x))))) 4)
(test (eval (parse `(let (x 3) (let (y 2) (+ x y))))) 5)
(test (eval (parse `(let (x 3) (* x (let (x 2) (+ x x)))))) 12)
